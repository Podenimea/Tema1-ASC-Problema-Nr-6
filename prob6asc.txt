;Se citeste o cifra a (0-9). Sa se determine daca este in multimea {0,2}.
section .data
	str: times 100 db 0 
	msg1	db	'Numarul apartine multimii {0,2}',0xa
	len1	equ	$ - msg1
	msg2	db	'Numarul NU apartine multimii {0,2}',0xa
	len2	equ	$ - msg2	

section .text

global _start

_start:
    
	mov	eax, 3          
	mov	ebx, 0          
	mov	ecx, str        
	mov	edx, 1       
	int	0x80 

	movzx	ecx, byte [str]
	
	sub	ecx, 48
	cmp	ecx, 0
	je	_et
	cmp	ecx, 2
	je	_et
	
	mov	edx, len2  
	mov	ecx, msg2    
	mov	ebx, 1	    
	mov	eax, 4	    
	int	0x80        
	
_exit:
	mov	eax, 1         
	mov	ebx, 0          
	int	0x80

_et:
	mov	edx, len1   
	mov	ecx, msg1   
	mov	ebx, 1	    
	mov	eax, 4	    
	int	0x80        
	
	jmp	_exit


#include <unistd.h>

char *mesaj1="Nu apartine!";
char *mesaj2="Apartine!";
char a[2];

void print(int n)
{
    n-=48;
    if(n==0||n==2)
        write(1, mesaj2, 9);
    else
        write(1, mesaj1, 12);
}

int main()
{
    read(0, a, 2);
    int tmp=0;
    tmp=*a;
    print(tmp);
    return 0;
}
